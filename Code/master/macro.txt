Sub DeleteRowsBasedOnColumnA()

Dim ws As Worksheet
Dim lastRow As Long
Dim i As Long
Dim cellValue As String

Set ws = ActiveSheet
lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

' Start from row 2 to keep header
For i = lastRow To 2 Step -1
    cellValue = LCase(Trim(ws.Cells(i, "A").Value))

    If cellValue = "" _
       Or InStr(cellValue, "from") > 0 _
       Or InStr(cellValue, "customer") > 0 _
       Or InStr(cellValue, "gl date") > 0 Then
       
        ws.Rows(i).Delete
    End If
Next i
End Sub


Sub SetPivotTableToClassic()
    Dim ws As Worksheet
    Dim pt As PivotTable

    ' Set the worksheet where your PivotTable is located
    Set ws = ThisWorkbook.Sheets("Export_pivot") ' Change "Sheet1" to your actual sheet name

    ' Loop through all PivotTables in the sheet
    For Each pt In ws.PivotTables
        pt.RowAxisLayout xlTabularRow ' Change layout to Classic (Tabular)
        pt.ShowTableStyleRowStripes = False ' Optional: Remove row striping
    Next

   
End Sub
